        if (!libro) {
            return 'Libro no encontrado';
        }
        
        if (!libro.disponible) {
            return 'El libro ya est√° prestado';
        }
        
        libro.disponible = false;
        return `Libro "${titulo}" prestado exitosamente`;
    }
};
