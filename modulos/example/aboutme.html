.about-me{
    display: flex;
    align-items: center;
    gap: 20px;

}

}

    overflow: hidden;

.about-me-image img{

    object-position: center;
}

.about-me-content h2 {
    font-size: 2rem;
}

    font-size: 1.1rem;
    margin-bottom: 20px;
}
.about-me-content .btn {
    font-size: 1rem;
